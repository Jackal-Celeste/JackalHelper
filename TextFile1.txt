module JackalHelperGrappleGem

using ..Ahorn, Maple

@mapdef Entity "JackalHelper/GrappleGem" GrappleGem(x::Integer, y::Integer)


const placements = Ahorn.PlacementDict(
    "Sombra Crystal (Jackal Helper)" => Ahorn.EntityPlacement(
        GrappleGem,
        "point",
    )
)


sprite = "characters/theoCrystal/idle00"


function Ahorn.selection(entity::GrappleGem)
    x, y = Ahorn.position(entity)
    return Ahorn.getSpriteRectangle(sprite, x, y)
end

Ahorn.render(ctx::Ahorn.Cairo.CairoContext, entity::GrappleGem, room::Maple.Room) = Ahorn.drawSprite(ctx, sprite, 0, 0)

end
